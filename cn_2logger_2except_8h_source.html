64#define CN_ENSURE(cond, lvl, e) \
65 do { \
66 enum CnTraceLvl _lvl = (lvl); \
67 \
68 if (CN_EXCEPTIONS_EN && !(cond)) { \
69 if (CN_ERROR == _lvl) { \
70 CN_TRACE( \
71 CN_ERROR, NULL, "%s:%d: %s", __FILE__, \
72 __LINE__, cn_except_##e.reason); \
73 cn_logger_cleanup(); \
74 cn_sysfail(); \
75 } else if (CN_WARNING == _lvl) { \
76 CN_TRACE( \
77 CN_WARNING, NULL, "%s:%d: %s", \
78 __FILE__, __LINE__, \

80 } else { \
81 CN_TRACE( \
82 CN_ERROR, NULL, "%s:%d: %s", __FILE__, \
83 __LINE__, \
84 cn_except_not_supported.reason); \
85 } \
86 } \
87 } while (0)
